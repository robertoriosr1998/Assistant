def parse_listings(html, base_url):
    soup = BeautifulSoup(html, "html.parser")
    jobs = []
    # This selector is heuristic; adjust if efc changes markup
    for card in soup.select("div.job-card, article.job-card, li.job-item"):
        title_el = card.select_one("h2, h3, a")
        title = title_el.get_text(strip=True) if title_el else None
        url = urljoin(base_url, title_el["href"]) if title_el and title_el.has_attr("href") else None

        company = None
        loc = None
        posted = None

        # Try common metadata
        company_el = card.select_one(".company, .employer, .brand")
        if company_el:
            company = company_el.get_text(strip=True)

        loc_el = card.select_one(".location, .city, .country")
        if loc_el:
            loc = loc_el.get_text(strip=True)

        posted_el = card.select_one(".posted, .date, time")
        if posted_el:
            posted = posted_el.get_text(strip=True)

        # Fallback: aggregate snippet/description if available
        snippet = None
        snippet_el = card.select_one(".description, .snippet, .summary")
        if snippet_el:
            snippet = snippet_el.get_text(strip=True)

        if title or url:
            jobs.append({
                "title": title,
                "company": company,
                "location": loc,
                "posted": posted,
                "url": url,
                "snippet": snippet
            })
    return jobs

    if not loc:
        return "", ""
    # Basic split by comma
    parts = [p.strip() for p in loc.split(",")]
    city = parts[0] if parts else ""
    country = parts[-1] if len(parts) > 1 else ""

def brave_discover(query, page=1):
    if not BRAVE_API_KEY:
        return []
    url = "https://api.brave.com/search"  # adjust if Brave's endpoint differs
    params = {"q": query, "count": 20, "page": page}
    headers = {"X-Api-Key": BRAVE_API_KEY}
    r = requests.get(url, headers=headers, params=params, timeout=20)
    r.raise_for_status()
    data = r.json()
    urls = []
    for item in data.get("web", []) + data.get("results", []):
        u = item.get("url")
        if u and "efinancialcareers.com" in u and ("in-europe" in u or "europe" in u or "/jobs/" in u):
